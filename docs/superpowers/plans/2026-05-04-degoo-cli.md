# degoo-cli Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a cross-platform Go CLI (`degoo upload` / `degoo download`) that syncs files to/from Degoo cloud storage with recursive directory support, timestamp-based skipping, 3-retry logic, and dual stdout+file logging.

**Architecture:** Cobra CLI wraps a sync engine that walks local/remote trees, compares `mtime` vs Degoo `LastModificationTime`, and calls a thin API client over Degoo's GraphQL + REST endpoints. Auth resolves credentials from `.env` beside the binary or interactive prompt, caches tokens in OS config dir.

**Tech Stack:** Go 1.22+, `github.com/spf13/cobra`, `github.com/joho/godotenv`, standard library (`net/http`, `encoding/json`, `crypto/sha1`, `encoding/base64`, `os`, `path/filepath`, `time`).

// internal/logger/logger.go
package logger

import (
	"fmt"
	"io"
	"os"
	"time"
)

type Logger struct {
	w io.Writer
	f *os.File
}

func New(logPath string) (*Logger, error) {
	f, err := os.OpenFile(logPath, os.O_APPEND|os.O_CREATE|os.O_WRONLY, 0644)
	if err != nil {
		return nil, err
	}
	return &Logger{w: io.MultiWriter(os.Stdout, f), f: f}, nil
}

func (l *Logger) Close() { l.f.Close() }

func (l *Logger) Info(format string, args ...interface{}) {
	l.write("INFO", format, args...)
}

func (l *Logger) Warn(format string, args ...interface{}) {
	l.write("WARN", format, args...)
}

func (l *Logger) Error(format string, args ...interface{}) {
	l.write("ERROR", format, args...)
}

func (l *Logger) write(level, format string, args ...interface{}) {
	ts := time.Now().Format("2006-01-02 15:04:05")
	msg := fmt.Sprintf(format, args...)
	fmt.Fprintf(l.w, "[%s] [%s] %s\n", ts, level, msg)
}
```

// internal/auth/auth.go
package auth

import (
	"bufio"
	"encoding/json"
	"errors"
	"fmt"
	"os"
	"path/filepath"
	"strings"

	"degoo-cli/internal/httpclient"
)

const (
	loginURL   = "https://rest-api.degoo.com/login"
	refreshURL = "https://rest-api.degoo.com/access-token/v2"
)

type Credentials struct {
	Email    string
	Password string
}

type TokenCache struct {
	AccessToken  string `json:"accessToken"`
	RefreshToken string `json:"refreshToken"`
}

func Login(creds Credentials) (*TokenCache, error) {
	body := map[string]interface{}{
		"Username":      creds.Email,
		"Password":      creds.Password,
		"GenerateToken": true,
	}
	var resp struct {
		Token        string `json:"Token"`
		RefreshToken string `json:"RefreshToken"`
	}
	if err := httpclient.PostJSON(loginURL, nil, body, &resp); err != nil {
		return nil, fmt.Errorf("login: %w", err)
	}
	if resp.Token == "" {
		return nil, errors.New("login: empty token in response")
	}

	// Exchange refresh token for access token
	var tokenResp struct {
		AccessToken string `json:"AccessToken"`
	}
	if err := httpclient.PostJSON(refreshURL, nil, map[string]string{"Token": resp.Token}, &tokenResp); err != nil {
		return nil, fmt.Errorf("token exchange: %w", err)
	}

	return &TokenCache{
		AccessToken:  tokenResp.AccessToken,
		RefreshToken: resp.Token,
	}, nil
}

func RefreshAccessToken(refreshToken string) (string, error) {
	var resp struct {
		AccessToken string `json:"AccessToken"`
	}
	if err := httpclient.PostJSON(refreshURL, nil, map[string]string{"Token": refreshToken}, &resp); err != nil {
		return "", fmt.Errorf("token refresh: %w", err)
	}
	return resp.AccessToken, nil
}

func cacheDir() (string, error) {
	dir, err := os.UserConfigDir()
	if err != nil {
		return "", err
	}
	p := filepath.Join(dir, "degoo-cli")
	return p, os.MkdirAll(p, 0700)
}

func cacheFile() (string, error) {
	dir, err := cacheDir()
	if err != nil {
		return "", err
	}
	return filepath.Join(dir, "keys.json"), nil
}

func SaveTokenCache(cache *TokenCache) error {
	path, err := cacheFile()
	if err != nil {
		return err
	}
	data, err := json.MarshalIndent(cache, "", "  ")
	if err != nil {
		return err
	}
	return os.WriteFile(path, data, 0600)
}

func LoadTokenCache() (*TokenCache, error) {
	path, err := cacheFile()
	if err != nil {
		return nil, err
	}
	data, err := os.ReadFile(path)
	if err != nil {
		return nil, err
	}
	var cache TokenCache
	return &cache, json.Unmarshal(data, &cache)
}

// ResolveCredentials reads from envPath (.env file), falls back to interactive prompt.
func ResolveCredentials(envPath string) (Credentials, error) {
	data, err := os.ReadFile(envPath)
	if err == nil {
		creds := parseEnv(string(data))
		if creds.Email != "" && creds.Password != "" {
			return creds, nil
		}
	}
	return promptCredentials()
}

func parseEnv(content string) Credentials {
	var c Credentials
	for _, line := range strings.Split(content, "\n") {
		line = strings.TrimSpace(line)
		if strings.HasPrefix(line, "USER=") {
			c.Email = strings.TrimPrefix(line, "USER=")
		}
		if strings.HasPrefix(line, "PASSWORD=") {
			c.Password = strings.TrimPrefix(line, "PASSWORD=")
		}
	}
	return c
}

func promptCredentials() (Credentials, error) {
	reader := bufio.NewReader(os.Stdin)
	fmt.Print("Degoo email: ")
	email, _ := reader.ReadString('\n')
	fmt.Print("Password: ")
	pass, _ := reader.ReadString('\n')
	return Credentials{
		Email:    strings.TrimSpace(email),
		Password: strings.TrimSpace(pass),
	}, nil
}
```

// internal/httpclient/httpclient.go
package httpclient

import (
	"bytes"
	"encoding/json"
	"fmt"
	"io"
	"net/http"
)

var httpClient = &http.Client{}

func PostJSON(url string, headers map[string]string, body, out interface{}) error {
	data, err := json.Marshal(body)
	if err != nil {
		return err
	}
	req, err := http.NewRequest(http.MethodPost, url, bytes.NewReader(data))
	if err != nil {
		return err
	}
	req.Header.Set("Content-Type", "application/json")
	for k, v := range headers {
		req.Header.Set(k, v)
	}
	resp, err := httpClient.Do(req)
	if err != nil {
		return err
	}
	defer resp.Body.Close()
	raw, _ := io.ReadAll(resp.Body)
	if resp.StatusCode >= 400 {
		return fmt.Errorf("HTTP %d: %s", resp.StatusCode, raw)
	}
	return json.Unmarshal(raw, out)
}

```go
// internal/api/client.go
package api

import (
	"fmt"
	"strings"
	"time"

	"degoo-cli/internal/httpclient"
)

const (
	graphqlURL = "https://production-appsync.degoo.com/graphql"
	apiKey     = "da2-vs6twz5vnjdavpqndtbzg3prra"
)

type FileInfo struct {
	ID           string
	Name         string
	FilePath     string
	Size         int64
	IsDirectory  bool
	ModifiedTime time.Time
	DownloadURL  string
}

type Client struct {
	token string
}

func NewClient(token string) *Client {
	return &Client{token: token}
}

func (c *Client) graphql(operationName, query string, variables map[string]interface{}, out interface{}) error {
	headers := map[string]string{
		"x-api-key": apiKey,
		"Token":     c.token,
	}
	body := map[string]interface{}{
		"operationName": operationName,
		"variables":     variables,
		"query":         query,
	}
	var wrapper struct {
		Data   interface{} `json:"data"`
		Errors []struct {
			Message string `json:"message"`
		} `json:"errors"`
	}
	wrapper.Data = out
	if err := httpclient.PostJSON(graphqlURL, headers, body, &wrapper); err != nil {
		return err
	}
	if len(wrapper.Errors) > 0 {
		return fmt.Errorf("graphql error: %s", wrapper.Errors[0].Message)
	}
	return nil
}

type rawNode struct {
	ID                   string  `json:"ID"`
	Name                 string  `json:"Name"`
	FilePath             string  `json:"FilePath"`
	Size                 int64   `json:"Size"`
	Category             int     `json:"Category"`
	IsFolder             bool    `json:"IsFolder"`
	LastModificationTime string  `json:"LastModificationTime"`
	URL                  string  `json:"URL"`
}

const childrenQuery = `
query getFileChildren5($ID: ID!, $Limit: Int, $Token: String) {
  getFileChildren5(ID: $ID, Limit: $Limit, Token: $Token) {
    Nodes {
      ID Name FilePath Size Category IsFolder LastModificationTime URL
    }
    NextToken
  }
}`

func (c *Client) GetChildren(folderID string) ([]FileInfo, error) {
	var all []FileInfo
	var nextToken *string

	for {
		vars := map[string]interface{}{
			"ID":    folderID,
			"Limit": 1000,
		}
		if nextToken != nil {
			vars["Token"] = *nextToken
		}

		var result struct {
			GetFileChildren5 struct {
				Nodes     []rawNode `json:"Nodes"`
				NextToken *string   `json:"NextToken"`
			} `json:"getFileChildren5"`
		}
		if err := c.graphql("getFileChildren5", childrenQuery, vars, &result); err != nil {
			return nil, err
		}

		for _, n := range result.GetFileChildren5.Nodes {
			all = append(all, toFileInfo(n))
		}

		nextToken = result.GetFileChildren5.NextToken
		if nextToken == nil || *nextToken == "" {
			break
		}
	}
	return all, nil
}

func toFileInfo(n rawNode) FileInfo {
	var mtime time.Time
	// Degoo returns milliseconds since epoch as string
	if n.LastModificationTime != "" {
		var ms int64
		fmt.Sscanf(n.LastModificationTime, "%d", &ms)
		mtime = time.UnixMilli(ms)
	}
	return FileInfo{
		ID:           n.ID,
		Name:         n.Name,
		FilePath:     n.FilePath,
		Size:         n.Size,
		IsDirectory:  n.IsFolder || n.Category == 4, // 4 = folder category
		ModifiedTime: mtime,
		DownloadURL:  n.URL,
	}
}

// ResolveRemotePath resolves a slash-separated path to a Degoo folder/file ID.
// "/" resolves to the root device folder.
func (c *Client) ResolveRemotePath(remotePath string) (string, error) {
	parts := splitPath(remotePath)

	// Root: get the first level children of ID "0" to find the device root
	children, err := c.GetChildren("0")
	if err != nil {
		return "", fmt.Errorf("resolve root: %w", err)
	}
	if len(parts) == 0 || (len(parts) == 1 && parts[0] == "") {
		if len(children) == 0 {
			return "0", nil
		}
		// Return first device root folder
		return children[0].ID, nil
	}

	currentID := "0"
	// First segment might match a device folder name or we skip it if it's root
	for _, part := range parts {
		if part == "" {
			continue
		}
		items, err := c.GetChildren(currentID)
		if err != nil {
			return "", fmt.Errorf("resolve %q: %w", part, err)
		}
		found := false
		for _, item := range items {
			if strings.EqualFold(item.Name, part) {
				currentID = item.ID
				found = true
				break
			}
		}
		if !found {
			return "", fmt.Errorf("path segment %q not found under ID %s", part, currentID)
		}
	}
	return currentID, nil
}

// EnsureRemotePath resolves a path, creating any missing folders along the way.
func (c *Client) EnsureRemotePath(remotePath string) (string, error) {
	parts := splitPath(remotePath)

	// Find device root
	children, err := c.GetChildren("0")
	if err != nil {
		return "", err
	}
	if len(parts) == 0 || (len(parts) == 1 && parts[0] == "") {
		if len(children) == 0 {
			return "0", nil
		}
		return children[0].ID, nil
	}

	currentID := children[0].ID // start from device root

	for _, part := range parts {
		if part == "" {
			continue
		}
		items, err := c.GetChildren(currentID)
		if err != nil {
			return "", err
		}
		found := ""
		for _, item := range items {
			if strings.EqualFold(item.Name, part) && item.IsDirectory {
				found = item.ID
				break
			}
		}
		if found != "" {
			currentID = found
			continue
		}
		newID, err := c.CreateFolder(currentID, part)
		if err != nil {
			return "", fmt.Errorf("create folder %q: %w", part, err)
		}
		currentID = newID
	}
	return currentID, nil
}

func splitPath(p string) []string {
	p = strings.TrimPrefix(p, "/")
	if p == "" {
		return nil
	}
	return strings.Split(p, "/")
}

type UploadRequest struct {
	ParentID string
	Name     string
	Data     []byte
	MTime    time.Time
}

// degooChecksum computes the Degoo-specific SHA1 checksum for file data.
// Format: base64(0x02 + sha1(0x02 + data))
func degooChecksum(data []byte) string {
	h := sha1.New()
	h.Write([]byte{0x02})
	h.Write(data)
	raw := h.Sum(nil)
	combined := append([]byte{0x02}, raw...)
	return base64.StdEncoding.EncodeToString(combined)
}

const uploadFileMutation = `
mutation setUploadFile3($ParentID: ID!, $Name: String!, $Size: Int!, $Checksum: String!, $LastModificationTime: String!) {
  setUploadFile3(ParentID: $ParentID, Name: $Name, Size: $Size, Checksum: $Checksum, LastModificationTime: $LastModificationTime) {
    UploadID
    Error
  }
}`

const bucketAuthQuery = `
query getBucketWriteAuth4($UploadID: ID!) {
  getBucketWriteAuth4(UploadID: $UploadID) {
    URL
    Fields { Key Value }
  }
}`

func (c *Client) UploadFile(req UploadRequest) error {
	checksum := degooChecksum(req.Data)
	mtimeMs := strconv.FormatInt(req.MTime.UnixMilli(), 10)

	var uploadResult struct {
		SetUploadFile3 struct {
			UploadID string `json:"UploadID"`
			Error    string `json:"Error"`
		} `json:"setUploadFile3"`
	}
	err := c.graphql("setUploadFile3", uploadFileMutation, map[string]interface{}{
		"ParentID":             req.ParentID,
		"Name":                 req.Name,
		"Size":                 len(req.Data),
		"Checksum":             checksum,
		"LastModificationTime": mtimeMs,
	}, &uploadResult)
	if err != nil {
		return fmt.Errorf("setUploadFile3: %w", err)
	}
	if uploadResult.SetUploadFile3.Error != "" {
		return fmt.Errorf("setUploadFile3 error: %s", uploadResult.SetUploadFile3.Error)
	}
	uploadID := uploadResult.SetUploadFile3.UploadID

	var authResult struct {
		GetBucketWriteAuth4 struct {
			URL    string `json:"URL"`
			Fields []struct {
				Key   string `json:"Key"`
				Value string `json:"Value"`
			} `json:"Fields"`
		} `json:"getBucketWriteAuth4"`
	}
	if err := c.graphql("getBucketWriteAuth4", bucketAuthQuery, map[string]interface{}{
		"UploadID": uploadID,
	}, &authResult); err != nil {
		return fmt.Errorf("getBucketWriteAuth4: %w", err)
	}

	return c.putToStorage(
		authResult.GetBucketWriteAuth4.URL,
		authResult.GetBucketWriteAuth4.Fields,
		req.Name,
		req.Data,
	)
}

func (c *Client) putToStorage(uploadURL string, fields []struct {
	Key   string `json:"Key"`
	Value string `json:"Value"`
}, filename string, data []byte) error {
	// Build multipart form with policy fields + file
	var buf bytes.Buffer
	boundary := "degoo-boundary-xyzabc"
	buf.WriteString("--" + boundary + "\r\n")
	for _, f := range fields {
		buf.WriteString(fmt.Sprintf("Content-Disposition: form-data; name=%q\r\n\r\n%s\r\n", f.Key, f.Value))
		buf.WriteString("--" + boundary + "\r\n")
	}
	buf.WriteString(fmt.Sprintf("Content-Disposition: form-data; name=\"file\"; filename=%q\r\n", filename))
	buf.WriteString("Content-Type: application/octet-stream\r\n\r\n")
	buf.Write(data)
	buf.WriteString("\r\n--" + boundary + "--\r\n")

	req, err := http.NewRequest(http.MethodPost, uploadURL, &buf)
	if err != nil {
		return err
	}
	req.Header.Set("Content-Type", "multipart/form-data; boundary="+boundary)

	resp, err := http.DefaultClient.Do(req)
	if err != nil {
		return err
	}
	defer resp.Body.Close()
	if resp.StatusCode >= 400 {
		body, _ := io.ReadAll(resp.Body)
		return fmt.Errorf("storage PUT HTTP %d: %s", resp.StatusCode, body)
	}
	return nil
}

// internal/sync/sync.go
package sync

import (
	"fmt"
	"os"
	"path/filepath"
	"time"

	"degoo-cli/internal/api"
	"degoo-cli/internal/logger"
)

type Stats struct {
	Uploaded int
	Skipped  int
	Failed   []string
	Bytes    int64
}

// ShouldTransfer returns true if the source (localMtime) is newer than the destination.
// Zero destination time means destination doesn't exist — always transfer.
func ShouldTransfer(sourceMtime, destMtime time.Time) bool {
	if destMtime.IsZero() {
		return true
	}
	return sourceMtime.After(destMtime)
}

// WithRetry calls fn up to maxAttempts times with exponential backoff (1s, 2s, 4s...).
func WithRetry(maxAttempts int, fn func() error) error {
	var err error
	for i := 0; i < maxAttempts; i++ {
		err = fn()
		if err == nil {
			return nil
		}
		if i < maxAttempts-1 {
			wait := time.Duration(1<<uint(i)) * time.Second
			time.Sleep(wait)
		}
	}
	return err
}

// Upload recursively uploads localPath to remotePath on Degoo.
func Upload(client *api.Client, log *logger.Logger, localPath, remotePath string) (*Stats, error) {
	stats := &Stats{}

	// Walk local filesystem
	type fileEntry struct {
		localFull  string
		localRel   string
		info       os.FileInfo
	}

	var files []fileEntry
	err := filepath.Walk(localPath, func(path string, info os.FileInfo, err error) error {
		if err != nil {
			return err
		}
		if info.IsDir() {
			return nil
		}
		rel, _ := filepath.Rel(localPath, path)
		files = append(files, fileEntry{path, rel, info})
		return nil
	})
	if err != nil {
		return nil, fmt.Errorf("walk %s: %w", localPath, err)
	}

	total := len(files)
	log.Info("Starting upload: %s → %s (%d files)", localPath, remotePath, total)

	for i, entry := range files {
		// Determine target remote dir
		remoteDir := remotePath
		relDir := filepath.Dir(entry.localRel)
		if relDir != "." {
			remoteDir = remotePath + "/" + filepath.ToSlash(relDir)
		}

		// Ensure remote directory exists
		parentID, err := client.EnsureRemotePath(remoteDir)
		if err != nil {
			log.Error("[%d/%d] Cannot create remote dir %s: %v", i+1, total, remoteDir, err)
			stats.Failed = append(stats.Failed, entry.localRel)
			continue
		}

		// Check remote file mtime
		children, _ := client.GetChildren(parentID)
		var remoteMtime time.Time
		for _, child := range children {
			if child.Name == entry.info.Name() {
				remoteMtime = child.ModifiedTime
				break
			}
		}

		if !ShouldTransfer(entry.info.ModTime(), remoteMtime) {
			log.Info("[%d/%d] Skipping %s (remote is up to date)", i+1, total, entry.localRel)
			stats.Skipped++
			continue
		}

		log.Info("[%d/%d] Uploading %s (%.1f KB)", i+1, total, entry.localRel, float64(entry.info.Size())/1024)

		data, err := os.ReadFile(entry.localFull)
		if err != nil {
			log.Error("[%d/%d] Cannot read %s: %v", i+1, total, entry.localRel, err)
			stats.Failed = append(stats.Failed, entry.localRel)
			continue
		}

		var uploadErr error
		for attempt := 1; attempt <= 3; attempt++ {
			uploadErr = client.UploadFile(api.UploadRequest{
				ParentID: parentID,
				Name:     entry.info.Name(),
				Data:     data,
				MTime:    entry.info.ModTime(),
			})
			if uploadErr == nil {
				break
			}
			if attempt < 3 {
				log.Warn("[%d/%d] Failed to upload %s (attempt %d/3), retrying...", i+1, total, entry.localRel, attempt)
				time.Sleep(time.Duration(1<<uint(attempt-1)) * time.Second)
			}
		}

		if uploadErr != nil {
			log.Error("[%d/%d] %s failed after 3 attempts: %v", i+1, total, entry.localRel, uploadErr)
			stats.Failed = append(stats.Failed, entry.localRel)
		} else {
			stats.Uploaded++
			stats.Bytes += entry.info.Size()
		}
	}

	return stats, nil
}

// Download recursively downloads remotePath from Degoo to localPath.
func Download(client *api.Client, log *logger.Logger, remotePath, localPath string) (*Stats, error) {
	stats := &Stats{}

	remoteID, err := client.ResolveRemotePath(remotePath)
	if err != nil {
		return nil, fmt.Errorf("resolve %s: %w", remotePath, err)
	}

	type remoteEntry struct {
		file      api.FileInfo
		localDest string
	}

	var queue []struct {
		id        string
		localBase string
	}
	queue = append(queue, struct {
		id        string
		localBase string
	}{remoteID, localPath})

	var files []remoteEntry

	for len(queue) > 0 {
		head := queue[0]
		queue = queue[1:]

		children, err := client.GetChildren(head.id)
		if err != nil {
			return nil, fmt.Errorf("list %s: %w", head.id, err)
		}
		for _, child := range children {
			dest := filepath.Join(head.localBase, child.Name)
			if child.IsDirectory {
				queue = append(queue, struct {
					id        string
					localBase string
				}{child.ID, dest})
			} else {
				files = append(files, remoteEntry{child, dest})
			}
		}
	}

	total := len(files)
	log.Info("Starting download: %s → %s (%d files)", remotePath, localPath, total)

	for i, entry := range files {
		localStat, statErr := os.Stat(entry.localDest)
		var localMtime time.Time
		if statErr == nil {
			localMtime = localStat.ModTime()
		}

		if !ShouldTransfer(entry.file.ModifiedTime, localMtime) {
			log.Info("[%d/%d] Skipping %s (local is up to date)", i+1, total, entry.file.Name)
			stats.Skipped++
			continue
		}

		log.Info("[%d/%d] Downloading %s (%.1f KB)", i+1, total, entry.file.Name, float64(entry.file.Size)/1024)

		var data []byte
		var downloadErr error
		for attempt := 1; attempt <= 3; attempt++ {
			data, downloadErr = client.DownloadFile(entry.file.DownloadURL)
			if downloadErr == nil {
				break
			}
			if attempt < 3 {
				log.Warn("[%d/%d] Failed to download %s (attempt %d/3), retrying...", i+1, total, entry.file.Name, attempt)
				time.Sleep(time.Duration(1<<uint(attempt-1)) * time.Second)
			}
		}

		if downloadErr != nil {
			log.Error("[%d/%d] %s failed after 3 attempts: %v", i+1, total, entry.file.Name, downloadErr)
			stats.Failed = append(stats.Failed, entry.file.Name)
			continue
		}

		if err := os.MkdirAll(filepath.Dir(entry.localDest), 0755); err != nil {
			log.Error("[%d/%d] Cannot create dir for %s: %v", i+1, total, entry.localDest, err)
			stats.Failed = append(stats.Failed, entry.file.Name)
			continue
		}
		if err := os.WriteFile(entry.localDest, data, 0644); err != nil {
			log.Error("[%d/%d] Cannot write %s: %v", i+1, total, entry.localDest, err)
			stats.Failed = append(stats.Failed, entry.file.Name)
			continue
		}

		stats.Downloaded++
		stats.Bytes += entry.file.Size
	}

	return stats, nil
}

		// Resolve defaults relative to executable
		exe, _ := os.Executable()
		exeDir := filepath.Dir(exe)

		if flagEnvPath == "" {
			flagEnvPath = filepath.Join(exeDir, ".env")
		}
		if flagLogPath == "" {
			flagLogPath = filepath.Join(exeDir, "degoo-cli.log")

		tokens, err := auth.Login(creds)

		}
		if err := auth.SaveTokenCache(tokens); err != nil {
			log.Warn("Could not save token cache: %v", err)
		}
		apiClient = api.NewClient(tokens.AccessToken)
		log.Info("Logged in as %s", creds.Email)
		return nil
	},

func printSummary(op string, stats *syncer.Stats) {
	log.Info("=== Transfer Summary ===")
	if op == "upload" {
		log.Info("Uploaded:  %d files (%.1f MB)", stats.Uploaded, float64(stats.Bytes)/1024/1024)
	} else {
		log.Info("Downloaded: %d files (%.1f MB)", stats.Downloaded, float64(stats.Bytes)/1024/1024)
	}
	log.Info("Skipped:   %d files (already up to date)", stats.Skipped)
	log.Info("Failed:    %d files", len(stats.Failed))
	for _, f := range stats.Failed {
		log.Info("  - %s  (max retries exceeded)", f)
	}
	log.Info("=======================")
}

func bytesToMB(b int64) string {
	return fmt.Sprintf("%.1f MB", float64(b)/1024/1024)
}

# .github/workflows/build.yml
name: Build

on:
  push:
    branches: [main]
    tags: ["v*.*.*"]
  pull_request:
    branches: [main]

jobs:
  build:
    runs-on: ubuntu-latest
    strategy:
      matrix:
        include:
          - goos: linux
            goarch: amd64
            name: degoo-cli-linux-amd64
          - goos: linux
            goarch: arm64
            name: degoo-cli-linux-arm64
          - goos: darwin
            goarch: amd64
            name: degoo-cli-darwin-amd64
          - goos: darwin
            goarch: arm64
            name: degoo-cli-darwin-arm64
          - goos: windows
            goarch: amd64
            name: degoo-cli-windows-amd64.exe
          - goos: windows
            goarch: arm64
            name: degoo-cli-windows-arm64.exe

    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-go@v5
        with:
          go-version: "1.22"

      - name: Build ${{ matrix.name }}
        env:
          GOOS: ${{ matrix.goos }}
          GOARCH: ${{ matrix.goarch }}
          CGO_ENABLED: "0"
        run: go build -ldflags="-s -w" -o dist/${{ matrix.name }} .

      - uses: actions/upload-artifact@v4
        with:
          name: ${{ matrix.name }}
          path: dist/${{ matrix.name }}

  release:
    if: startsWith(github.ref, 'refs/tags/')
    needs: build
    runs-on: ubuntu-latest
    permissions:
      contents: write
    steps:
      - uses: actions/download-artifact@v4
        with:
          path: dist
          merge-multiple: true

      - uses: softprops/action-gh-release@v2
        with:
          files: dist/*
```
